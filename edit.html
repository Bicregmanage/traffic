async function fetchEntry(number) { try { const response = await fetch(`https://shqipfhr9d.execute-api.us-east-2.amazonaws.com/entries?number=${number}`); const entries = await response.json(); const entry = entries.find(e => e.number === number); if (entry) { document.getElementById("numberInput").value = entry.number; document.getElementById("nameInput").value = entry.name; document.getElementById("memoInput").value = entry.memo; } else { alert("エントリが見つかりませんでした"); } } catch (error) { console.error("読み込みエラー:", error); alert("エントリの読み込みに失敗しました"); } }
